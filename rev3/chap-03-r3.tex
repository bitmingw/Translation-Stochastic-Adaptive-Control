\chapter{线性随机系统的性质} % Properties of Linear Stochastic Systems

本章简要介绍了含有输入干扰和测量误差的线性随机系统模型，并重点讨论了当干扰是高斯噪声时的情况。本章推导了状态和测量过程的平均值和协方差和递推公式。递推中的渐近性质和状态方程中的稳定性密切相关。最后，本章说明线性高斯系统的状态和高斯-马尔可夫过程之间存在联系。

\section{线性高斯系统}

对于n维随机变量$z$，
\begin{equation}
	z \sim N(\overline{z},\Sigma) \quad or \quad 
	p(z) \sim N(\overline{z},\Sigma) \label{eq:3.1.1}
\end{equation}
指出$z$服从平均值为$\overline{z}$，协方差为$\Sigma$的高斯分布。\\
其中$z$的概率分布是
\begin{equation}
	p(z) = [(2 \pi)^n \ \vert \Sigma \vert]^{{-1}/2} exp[-\frac{1}{2} (z - \overline{z})^T \Sigma^{-1} (z - \overline{z})] \label{eq:3.1.2}
\end{equation}
式中$\vert \Sigma \vert := det \ \Sigma$。\\

对于线性随机系统
\begin{subequations} \label{eq:3.1.3}
	\begin{align}
		x_{k+1} &= A x_k + B u_k + G w_k \label{eq:3.1.3a} \\
		y_k &= C x_k + H v_k \label{eq:3.1.3b}
	\end{align}
\end{subequations}
式中$x_k \in R^n$，$u_k \in R^m$，$y_k \in R^p$，$w_k \in R^g$，$v_k \in R^h$且$A$，$B$，$G$，$C$，$H$为相应大小的固定矩阵。假设独立随机变量$x_0, w_0, w_1,..., v_0, v_1,...$均服从高斯分布：
\begin{equation}
	x_0 \sim N(\overline{x}_0,\Sigma_0), \quad
	w_k \sim N(0,Q), \quad
	v_k \sim N(0,R). \label{eq:3.1.4}
\end{equation}

为了计算该系统的一步转移概率，我们可以直接运用公式\eqref{eq:2.6.7}，但从最初的原理开始可以获得更有建设性的解释。因为$w_k$服从高斯分布，$G w_k$也服从高斯分布。其平均值为$E \ G  w_k  =  G \ E \ w_k = 0$，其协方差为$E \ (G w_k)(G w_k)^T = G \ E \ w_k {w_k}^T G^T  =  GQG^T$。因此$G w_k$服从$G w_k \sim N(0,GQG^T)$。

% ==== End of Page 23 ====

因为假设上述的随机变量是相互独立的，所以对于任意反馈法则$g$，$w_k$与$x_k$和$u_k$相互独立。从\eqref{eq:3.1.3a}可以看出，在已知$x_k$和$u_k$时$x_{k+1}$的条件概率分布等同于$G  w_k$的概率分布的平均值偏移了$A x_k + B u_k$。因此系统的状态转移概率是
\begin{equation}
	p(x_{k+1} \ \vert \ x_k,u_k) \sim N(A x_k + B u_k, GQG^T). \label{eq:3.1.5}
\end{equation}
代入式\eqref{eq:3.1.2}可得表达式
\begin{align*}
	p(x_{k+1} \ \vert \ x_k, u_k) = & K \ exp[-\frac{1}{2} (x_{k+1} - A x_k - B u_k)^T \\
	& \times (GQG^T)^{-1} (x_{k+1} - A x_k - B u_k)]
\end{align*}
式中的$K$为归一化常数
\begin{equation*}
	K = [(2 \pi)^n \ \vert GQG^T \vert]^{{-1}/2}.
\end{equation*}

类似地可以得到输出概率（观测概率）为
\begin{equation}
	p(y_k \ \vert \ x_k) \sim N(C x_k, HRH^T). \label{eq:3.1.6}
\end{equation}

下面计算系统的$m$步转移概率。设$g$为满足$g_{k+1} \equiv u_{k+1}, ..., g_{k+m-1} \equiv u_{k+m-1}$且$g_{k+1}, ..., g_{k+m-1}$为常值函数的任何反馈法则。设$\{x_k\},\{y_k\}$为相应的系统状态和输出过程。由定义\eqref{Def:2.6.14}可知$m$步转移概率是已知$x_k,u_k, ..., u_{k+m-1}$时$x_{k+m}$的条件概率，它与反馈法则$g$无关。由定义\eqref{Def:2.6.14}
可以用更简单的方式直接计算此概率。从公式\eqref{eq:3.1.3a}可得
\begin{equation}
	x_{k+m} = A^m x_k + \sum_{j=0}^{m-1} A^{m-1-j} B u_{k+j} 
	+ \sum_{j=0}^{m-1} A^{m-1-j} G w_{k+j}. \label{eq:3.1.7}
\end{equation}
因为随机变量$\{w_{k+j}, j \ge 0\}$服从高斯分布，且与$x_k$，$u_k$，$u_{k+j}, j \ge 1$相互独立，故从公式\eqref{eq:3.1.7}可知
\begin{equation}
	p(x_{k+m} \ \vert \ x_k,u_k,...,u_{k+m-1}) \sim
	N(A^m x_k + \sum_{j=0}^{m-1} A^{m-1-j} B u_{k+j}, \Sigma_{k+m \vert k}) \label{eq:3.1.8}
\end{equation}
式中$\Sigma_{k+m \vert k} := cov(\sum_{j=0}^{m-1} A^{m-1-j} G w_{k+j})$。这个协方差的计算与$w_k$的值无关：
	\begin{align}
		\Sigma_{k+m \vert k} &= cov(\sum_{j=0}^{m-2} A^{m-1-j} G w_{k+j}) + cov(G w_{k+m-1}) \nonumber \\
		% ==== End of Page 24 ====
		&= cov(A \sum_{j=0}^{m-2} A^{m-2-j} G w_{k+j}) + GQG^T \nonumber \\
		&= A [cov(\sum_{j=0}^{m-2} G w_{k+j})] A^T + GQG^T \nonumber \\
		&= A \Sigma_{k+m-1 \vert k} A^T + GQG^T \label{eq:3.1.9}
	\end{align}
通过初始条件
\begin{equation}
	\Sigma_{k \vert k} = 0 \label{eq:3.1.10}
\end{equation}
可以递归求得所有$\Sigma_{k+m \vert k}, m > 0$。
从公式\eqref{eq:3.1.9}和\eqref{eq:3.1.10}可以看出$\Sigma_{k+m \vert k}$只取决于$m$，而与$k$的值无关。

注意上述讨论的反馈函数$g_1,...,g_k$（而不是$g_i, i \ge k$）并不需要为常值函数。事实上，$g_i, i \le k$可以为任何非线性函数。 % of the available information \{y^i\}
如果反馈函数是非线性的，那么控制过程也将是非高斯的，因此由无条件概率决定的状态过程也将是非高斯的。即便如此，由条件概率决定的转移概率依旧服从高斯分布。这就构成了下面的引理。
\begin{Lem} \label{lem:3.1.11}
	由公式\eqref{eq:3.1.3}和\eqref{eq:3.1.4}所描述的$m$步转移概率，当反馈法则$g$满足$g_{k+1} \equiv u_{k+1},...,g_{k+m-1} \equiv u_{k+m-1}$时，服从高斯分布：
	\begin{equation}
		p(x_{k+m} \ \vert \ x_k,u_k,...,u_{k+m-1}) \sim N(A^m \ x_k + \sum_{j=0}^{m-1} A^{m-1-j} B \ u_{k+j}, \Sigma_{k+m \vert k})
	\end{equation}
	此外，协方差矩阵$\Sigma_{k+m \vert k}$可以通过线性差分方程\eqref{eq:3.1.9}和\eqref{eq:3.1.10}算出。
\end{Lem}

假设我们的控制率总是开环的，即$g_i \equiv u_i$对$i \ge 0$和$i \le k$总是常值函数。这时状态过程$\{x_k\}$服从高斯分布。它的无条件概率分布完全由无条件的均值$E \ x_k^g$和无条件的协方差$\Sigma_{k+m \vert k} := cov(x_{k+m},x_k)$描述。

概率$p_{x_k}^g$决定了均值$E \ x_k^g$和方差$\Sigma_{k,k} := cov(x_k,x_k) := \Sigma_k$。注意到
\begin{equation*}
	x_k = A^k x_0 + \sum_{j=0}^{k-1} A^{k-1-j} B u_j + \sum_{j=0}^{k-1} A^{k-1-j} G w_j
\end{equation*}
% ==== End of Page 25 ====
右边的三项相互独立且满足高斯分布，所以
\begin{equation*}
	p_{x_k}^g \sim N(\overline{x}_k, \Sigma_k)
\end{equation*}
其中
\begin{align}
	\overline{x}_k &= A^k \overline{x}_0 + \sum_{j=0}^{k-1} A^{k-1-j} B u_j, \nonumber \\
	\Sigma_k &= cov(A^k x_0 + \sum_{j=0}^{k-1} A^{k-1-j} G w_j) \nonumber \\
	&= cov(A(A^{k-1} x_0 + \sum_{j=0}^{k-2} A^{k-2-j} G w_j)) + cov(G w_{k-1}) \nonumber \\
	&= A \Sigma_{k-1} A^T + GQG^T, \quad k \ge 1, \label{eq:3.1.12} \\
	\Sigma_0 &= cov(x_0). \label{eq:3.1.13}
\end{align}
因此$\{x_k\}$的概率分布可以递归地计算出来。

为了计算协方差$\Sigma_{k+m,k}$的值，从它的定义出发
\begin{equation*}
	\Sigma_{k+m,k} := E \ (x_{k+m} - \overline{x}_{k+m}) (x_k - \overline{x}_k)^T.
\end{equation*}
从公式\eqref{eq:3.1.7}可得
\begin{equation*}
	x_{k+m} - \overline{x}_{k+m} = A^m (x_k - \overline{x}_k) + \sum_{j=0}^{m-1} A^{m-1-j} G w_{k+j}
\end{equation*}
因为$x_k$与$w_{k+j}, j \ge 0$相互独立，所以
\begin{equation}
	\Sigma_{k+m,k} = A^m \Sigma_{k}. \label{eq:3.1.14}
\end{equation}
注意到均值$\overline{x}_k$只由控制过程$\{u_k\}$决定，但协方差$\Sigma_{k+m,k}$受到更多的因素影响。

当$g$是开环时，过程$y_k$服从高斯分布。从公式$\eqref{eq:3.1.3b}$可以看出
\begin{equation*}
	p_{y_k}^g \sim N(\overline{y}_k, \Sigma_{k}^{y})
\end{equation*}
其中均值为
\begin{equation*}
	\overline{y}_k := E \ y_k = C \ E \ x_k = C (A^k \overline{x}_0 + \sum_{j=0}^{k-1} A^{k-1-j} B u_j)
\end{equation*}
方差为
\begin{equation}
	\Sigma_k^y = cov(C x_k) + cov(H v_k) = C \Sigma_k C^T + HRH^T. \label{eq:3.1.15}
\end{equation}
同时可以得到$y_{k+m}$和$y_k$的方差。
% ==== End of Page 26 ====
$y_{k+m}$的方差定义为
\begin{equation*}
	\Sigma_{k+m,k}^y := E \ (y_{k+m} - \overline{y}_{k+m}) (y_k - \overline{y}_k)^T.
\end{equation*}
由公式\eqref{eq:3.1.3b}和公式\eqref{eq:3.1.14}，考虑到$\{x_k\}$和$\{v_k\}$之间的独立性，可得
\begin{align}
	\Sigma_{k+m,k}^y &= C \Sigma_{k+m,k} C^T, \quad m \ge 1, \nonumber \\
	&= C A^m \Sigma_k C^T, \quad m \ge 1. \label{eq:3.1.16}
\end{align}
下面我们通过一个练习检测时变线性随机系统。

\begin{Exe} \label{Exe:3.1.17}
除了公式\eqref{eq:3.1.3}和\eqref{eq:3.1.4}的非时变系统之外，也有如下的时变系统
\begin{align*}
	x_{k+1} &= A^k x_k + B^k u_k + G_k w_k, \\
	y_k &= C_k x_k + H_k v_k,
\end{align*}
且独立变量服从高斯分布
\begin{equation*}
	x_0 \sim N(\overline{x}_0, \Sigma_0), \quad 
	w_k \sim N(0, Q_k), \quad 
	v_k \sim N(0, R_k).
\end{equation*}
请证明$m$步转移概率也服从高斯分布
\begin{align*}
	&p(x_{k+m} \ \vert \ x_k,u_k,...,u_{k+m-1}) \sim \\
	&N(A_{k+m-1} ... A_k  x_k + \sum_{j=0}^{m-1} A_{k+m-1} ... A_{k+j+1} B_{k+j} u_{k+j}, \Sigma_{k+m \vert k}).
\end{align*}
并且高斯分布的协方差可以使用时变线性差分方程计算
\begin{align*}
	\Sigma_{k+m \vert k} &= A_{k+m-1} \Sigma_{k+m-1 \vert k} A_{k+m-1}^T + G_{k+m-1} Q_{k+m-1} G_{k+m-1}^T, \\
	\Sigma_{k \vert k} &= 0.
\end{align*}
\end{Exe}

在上文中我们均假设系统的输入噪声$\{w_k\}$是一个独立过程。下面的练习将展示如何通过扩大状态向量消除这个假设。

\begin{Exe} \label{Exe:3.1.18}
假设噪声序列$\{w_0,w_1,...\}$并不是相互独立的。因此系统\eqref{eq:3.1.3},\eqref{eq:3.1.4}不满足线性随机系统的定义。但是我们假设$\{w_k\}$本身是一个线性系统的输出
\begin{equation*}
	\xi_{k+1} = F \xi_k + \epsilon_k; \quad w_k = D \xi_k + \delta_k,
\end{equation*}
其中变量$\{x_0,\xi_0,...,v_0,...,\epsilon_0,...,\delta_0,...$都相互独立且服从高斯分布。证明$\{y_k\}$可以表示为一个线性随机系统。
\textnormal{[}提示：可将状态向量改写为$\zeta_k := (x_k^T,\xi_k^T)^T$\textnormal{]}
\end{Exe}

因此线性随机系统的定义可以扩大到系统输入噪声为经过线性随机系统过滤后的高斯白噪声的情形。

\section{线性非高斯系统}

对于公式\eqref{eq:3.1.3}所示的线性系统，假设独立随机变量$x_0,w_0,w_1,...,v_0,v_1,...$并不服从高斯分布，但它们的均值和方差仍和先前相同：
\begin{align*}
	E \ x_0 = \overline{x}_0, \quad E \ w_k = 0, \quad E \ v_k = 0, \\
	cov(x_0) = \Sigma_0, \quad cov(w_k) = Q, \quad cov(v_k) = R.
\end{align*}
与此同时，使用开环控制率使得$g_k = u_k, k \ge 0$是常值函数。
设
\begin{align*}
	\overline{x}_{k+m \vert k} := E \ \{x_{k+m} \vert x_k,u_k,...,u_{k+m-1}\}, \\
	\Sigma_{k+m \vert k} := E \ \{(x_{k+m} - \overline{x}_{k+m \vert k}) (x_{k+m} - \overline{x}_{k+m \vert k})^T \vert x_k,u_k,...,u_{k+m-1}\},
\end{align*}
在已知$x_k$时$x_{k+m}$的条件均值和条件概率。

从公式\eqref{eq:3.1.7}可知
\begin{align*}
	\overline{x}_{k+m \vert k} = A^m x_k + \sum_{j=0}^{m-1} A^{m-1-j} B u_{k+j}, \\
	\Sigma_{k+m \vert k} = cov(\sum_{j=0}^{m-1} A^{m-1-j} G w_{k+j}),
\end{align*}
展开可得
\begin{equation*}
	\Sigma_{k+m \vert k} = A \Sigma_{k+m-1 \vert k} A^T + GQG^T.
\end{equation*}
与公式\eqref{eq:3.1.8}相比，前两项条件矩和高斯系统的情形相同，但条件概率分布并不一定服从高斯分布。

类似地，当系统是开环时，无条件均值和协方差
\begin{equation*}
\overline{x}_k := E \ x_k, \quad \Sigma_k := E \ (x_k - \overline{x}_k) (x_k - \overline{x}_k)^T
\end{equation*}
也由公式\eqref{eq:3.1.12}给出。

通过下面的练习，我们可以看出，前两项矩恰好相同的情况对练习\eqref{Exe:3.1.17}中的时变系统也成立。

\section{渐进性质}

下面我们研究当$k \to \infty$时由公式\eqref{eq:3.1.12}和\eqref{eq:3.1.13}给定的协方差矩阵$\Sigma_k$的变化。$\Sigma_k$的计算公式如下：
\begin{align}
	\Sigma_k &= A \Sigma_{k-1} A^T + GQG^T, \quad k \ge 1, \label{eq:3.3.1} \\
	\Sigma_0 &= cov(x_0) \label{eq:3.3.2}
\end{align}
将递归公式展开后，可以得到
\begin{equation}
	\Sigma_k = A^k \Sigma_0 (A^k)^T + \sum_{j=0}^{k-1} A^j GQG^T (A^j)^T. \label{eq:3.3.3}
\end{equation}
为了得到收敛的$\Sigma_k$序列，整个级数需要可求和，因此矩阵$A^j$要收敛到0.

如果矩阵$A$是稳定的，那么它的所有特征值的绝对值严格小于1。

\begin{Thm} \label{Thm:3.3.4}
假设矩阵$A$是稳定的，则存在一个半正定矩阵$\Sigma_{\infty}$使得$\lim_{k \to \infty} \Sigma_k = \Sigma \infty$. 并且，$\Sigma_{\infty}$是下面线性方程的唯一解：
\begin{equation}
	\Sigma_{\infty} = A \Sigma_{\infty} A^T + GQG^T. \label{eq:3.3.5}
\end{equation}
\end{Thm}

\begin{Prf}
因为$A$是稳定的，所以存在数$K < \infty$和$0 < \alpha < 1$使$A^j$的模小于$K \ \alpha^j$对任意$A$的参数都成立。因此公式\eqref{eq:3.3.3}中的第一项消失了，且整个级数可以求和。可以得到
\begin{equation*}
	\lim_{k \to \infty} \Sigma_k = \Sigma_{\infty} := \sum_{j=0}^{\infty} A^j GQG^T (A^j)^T.
\end{equation*}
接下来，由于$\Sigma_k$收敛至$\Sigma_{\infty}$，我们让$k \to \infty$，那么$\Sigma_{\infty}$服从公式\eqref{eq:3.3.5}。因此，我们只需要证明公式\eqref{eq:3.3.5}有唯一解。假设$\Sigma_{\infty}^1$和$\Sigma_{\infty}^2$是\eqref{eq:3.3.5}的两个解。那么他们的差值$\Delta := \Sigma_{\infty}^1 - \Sigma_{\infty}^2$满足
\begin{equation*}
	\Delta \ = \ A \Delta A^T.
\end{equation*}
迭代带入此式，可得
\begin{equation*}
	\Delta \ = \ A^k \Delta (A^k)^T
\end{equation*}
当$k \to \infty$时显然有$\Delta = 0$，因此$\Sigma_{\infty}^1 = \Sigma_{\infty}^2$，证毕。
\end{Prf}
% ==== End of Page 29 ====
公式\eqref{eq:3.3.5}称为离散李雅普诺夫方程。

\begin{Exe} \label{Exe:3.3.6}
假设$A$是稳定的，若$cov(x_0) = \Sigma_{\infty}$，则$\{x_k\}$的协方差函数是平稳的。即$\Sigma_k = \Sigma_{\infty}$且$\Sigma_{k+m,k} = A^m \Sigma_{\infty}$对所有$k > 0$均成立。一般而言，$\{x_k\}$可以被分解成两部分
\begin{equation*}
	x_k = x_k' + x_k''
\end{equation*}
其中$\{x_k'\}$的协方差函数是平稳的 % TO BE CHECKED
，且当$k \to \infty$时$E \ \vert x_k'' \vert^2 \to 0$.（分解方式可能不唯一。）
\end{Exe}

\begin{Rem} \label{Rem:3.3.7}
某些时候，即使$A$不是稳定的，$\Sigma_k$仍旧会收敛。一个简单的例子是当$\Sigma_0 = 0$，$Q = 0$时，$\Sigma_k \equiv 0$而与$A$无关。另外，$\Sigma_{\infty}$也不一定是严格正定的。例如当$A = 0$且$GQG^T < n$的时候。上面的两个例子说明，如果输入扰动$\{w_k\}$影响了状态向量的每个组成部分，那么$A$的稳定性必须满足收敛到$\Sigma_k$，且协方差的极限$\Sigma_{\infty}$是正定的。如果合适地定义了影响所有组成部分的标记，上面的说法是很有说服力的。
\end{Rem}

一对维度分别为$n \times n$和$n \times g$的矩阵$A$和$S$，当维度为$n \times ng$的矩阵$[S \ AS \ A^{n-1}S]$的秩为$n$时，则成这个状态是可达的。可达性将通过下面的习题证明。

\begin{Exe} \label{Exe:3.3.8}
证明下面的几种说法相互等价: \label{3.3.8} \\ % Should be defined later
(1) $A$，$S$是一个可达对
(2) $n \times n$的矩阵$\sum_{j=0}^{n-1} A^j SS^T (A^j)^T$是正定的
(3) 对任意$x \in \mathbb{R}^n$，存在一个输入序列$w_0,...,w_{n-1}$属于$\mathbb{R}^g$，使下面的确定性线性系统
\begin{equation*}
	x_{k+1} = A \ x_k + S \ w_k, \quad k \ge 0
\end{equation*}
从状态$x_0 = 0$转移到$x_n = x$。
\end{Exe}

\begin{Thm} \label{Thm:3.3.9}
设$A$，$S$是一个可达对，则下面的说法相互等价： \\
(1) $A$是稳定的。
(2) 方程
\begin{equation}
	\Sigma = A \Sigma A^T + SS^T \label{eq:3.3.10}
\end{equation}
有一个正定的解$\Sigma$。
\end{Thm}

\begin{Prf}
如果$A$是稳定的，由定理\eqref{Thm:3.3.4}可知，存在一个唯一的半正定解，
% ==== End of Page 30 ====
\begin{equation*}
	\Sigma = \sum_{j=0}^{\infty} A^j SS^T (A^j)^T.
\end{equation*}
因为$A$，$S$是可达的，由练习\eqref{Exe:3.3.8}可得$\Sigma$是正定的。

现在假设$\Sigma$是公式\eqref{eq:3.3.10}的一个正定解，那么有
\begin{equation}
	\Sigma = A^k \Sigma (A^k)^T + \sum_{j=0}^{k-1} A^j SS^T (A^j)^T, \quad k \ge 0. \label{eq:3.3.11}
\end{equation}
如果$\lambda$是$A$的一个特征值，并且有一个向量$x \neq 0$满足$A^T \ x = \lambda x$。在式\eqref{eq:3.3.11}左乘$x^{\ast}$（$\ast$代表共轭转置），右乘$x$，另$k = n-1$时，有
\begin{equation*}
	x^{\ast} \Sigma x = \vert \lambda \vert^{2(n-1)} x^{\ast} \Sigma x + x^{\ast} [\sum_{j=0}^{n-1} A^j SS^T (A^j)^T] x.
\end{equation*}
括号$[...]$中的矩阵由练习\eqref{Exe:3.3.8}可知是正定的，因此上式中的最后一项是严格大于0的。最后可以得出$\vert \lambda \vert < 1$并且$A$是稳定的。
\end{Prf}

回到协方差公式\eqref{eq:3.3.1}，若矩阵$S$满足
\begin{equation}
	GQG^T = SS^T \label{eq:3.3.12}
\end{equation}
则称$S$是$GQG^T$的一个平方根。

\begin{Thm} \label{Thm:3.3.13}
假设$A$，$S$是可达的，则下面的命题是等价的：
(1) $A$是稳定的。
(2) 当$k \to \infty$时$\Sigma_k$收敛至正定矩阵$\Sigma_{\infty}$。
\end{Thm}

\begin{Prf}
由定理\eqref{Thm:3.3.9}直接可得。
\end{Prf}

\begin{Rem}
式\eqref{eq:3.3.12}中的平方根$S$并不是唯一的。但是从练习\eqref{Exe:3.3.8}
可知，$A$，$S$是可达的当且仅当$\sum_{j=0}^{n-1} A^j SS^T (A^j)^T = \sum_{j=0}^{n-1} A^j GQG^T (A^j)^T$是正定的。因此$A$，$S$的可达性并不取决于\eqref{eq:3.3.12}中平方根的选取。
\end{Rem}

\section{高斯-马尔可夫过程}

如果一个在$\mathbb{R}^n$内取值的随机过程$\{x_k,k \ge 0\}$是高斯的，那么称这个随机过程为高斯-马尔可夫过程。也就是说，对任意$k$，随机变量$x_0,...x_k$的概率分布服从高斯分布，并且满足马尔可夫性质：
\begin{equation}
	p(x_{k+1} \vert x_k,...,x_0) = p(x_{k+1} \vert x_k) \label{eq:3.4.1}
\end{equation}

% ==== End of page 31 ====

我们已经知道，状态过程$\{x_k\}$由
\begin{equation*}
	x_{k+1} = A_k \ x_k + w_k
\end{equation*}
给出。如果$x_0,w_0,w_1,...$是相互独立的且服从高斯分布，那么这个过程就是高斯-马尔可夫过程。我们把这个这个命题的逆命题作为练习。假设$\{x_k\}$是一个高斯-马尔可夫过程，$E \ x_k \equiv 0$，指定$\Sigma_k := cov(x_k)$，$\Sigma_{k+1,k} := cov(x_{k+1},x_k)$。

\begin{Exe} \label{Exe:3.4.2}
试证明
\begin{align}
	\hat{x}_{k+1 \vert k} :=& E \ \{x_{k+1} \vert x_k,...,x_0\} = E \ \{x_{k+1} \vert x_k\} \nonumber \\
	=& \Sigma_{k+1,k} \Sigma_{k}^{-1} x_k. \label{eq:3.4.3}
\end{align}
若令
\begin{equation*}
	w_k := x_{k+1} - \hat{x}_{k+1 \vert k}
\end{equation*}
试证明对高斯分布且相互独立的随机变量$x_0,w_0,w_1,...$，高斯-马尔可夫过程可以表示为
\begin{equation}
	x_{k+1} = A_k \ x_k + w_k \label{eq:3.4.4}
\end{equation}
其中$A_k := \Sigma_{k+1,k} \Sigma_k^{-1}$。
\textnormal{[}提示：\eqref{eq:3.4.3}的第一个等式可以由马尔可夫的性质\eqref{eq:3.4.1}得到。它的第二个等式和剩下的部分，可以通过计算$x_{k+1} - \Sigma_{k+1,k} \Sigma_k^{-1} x_k$和$x_k$之间的协方差证明他们之间没有相关关系。由于两个耦合的高斯随机变量是非相关的，所以他们是相互独立的，也就是说$E \ \{\hat{x}_{k+1} - \Sigma_{k+1,k} \Sigma_{k}^{-1} x_k \vert x_k = 0$，因此\eqref{eq:3.4.3}的第二项也得到了验证。\textnormal{]}
\end{Exe}

在上面的练习中，$w_k = x_{k+1} - \hat{x}_{k+1 \vert k}$和$x_k$共同决定了$x_{k+1}$。$x_k$这一项可以在$k$时刻知道，但$w_k$与先前的历史无关，因为无法在$k$时刻获知。因此$w_k := \hat{x}_{k+1 \vert k} - x_{k+1}$又称为新型，或者说是过去未出现的关于$x_{k+1}$的新信息。式\eqref{eq:3.4.4}也称为过程$\{x_k\}$的新型表示。
